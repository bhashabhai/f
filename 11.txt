    const http = require('http');
    const port = 3000;

    const server = http.createServer((req, res) => {
      res.writeHead(200, { 'Content-Type': 'text/plain' });
      res.end('ping\n');
    });

    server.listen(port, () => {
      console.log(`Server running on port ${port}`);
    });

RUN npm install

# Copy the rest of the app code
COPY . .

# Expose the port
EXPOSE 3000

# Start the server
