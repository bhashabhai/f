fetch("https://raw.githubusercontent.com/bhashabhai/FSWD/main/users.json")
  .then(r => r.json())
  .then(d => {
    let x = "";
    d.forEach(e => {
      x += `<div>
        <b>${e.Name}</b><br>
        Office: ${e.Office}<br>
        Position: ${e.Position}<br>
        Salary: ${e.Salary}
        <hr>
      </div>`;
    });
    document.getElementById("c").innerHTML = x;
  });
